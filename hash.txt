import hashlib

fileDict = {"files":[], "fileHashes":[]}

def hashedDict(file, md5, sha1):
    fileDict["files"].append([file])
    fileDict["fileHashes"].append([md5, sha1])
    print(fileDict)

def counter(file):
    count = 0
    for char in file:
        count = count + 1
    return(count)

    dot = "."
